#include <stdio.h>

void find(int *a, int n, int *max, int *min) {
    *max = *min = a[0];
    for(int i = 1; i < n; i++) {
        if(a[i] > *max) *max = a[i];
        if(a[i] < *min) *min = a[i];
    }
}

int main() {
    int arr[5] = {2, 8, 1, 6, 3};
    int max, min;
    find(arr, 5, &max, &min);
    printf("Max = %d, Min = %d", max, min);
    return 0;
}
