#include <stdio.h>
#include <stdlib.h>

int main() {
    int r = 2, c = 2;

    int **A, **B, **C;

    A = (int**)malloc(r * sizeof(int*));
    B = (int**)malloc(r * sizeof(int*));
    C = (int**)malloc(r * sizeof(int*));

    for(int i = 0; i < r; i++) {
        A[i] = (int*)malloc(c * sizeof(int));
        B[i] = (int*)malloc(c * sizeof(int));
        C[i] = (int*)malloc(c * sizeof(int));
    }

    for(int i = 0; i < r; i++)
        for(int j = 0; j < c; j++) {
            scanf("%d", &A[i][j]);
            scanf("%d", &B[i][j]);
            C[i][j] = A[i][j] - B[i][j];
        }

    printf("Result:\n");
    for(int i = 0; i < r; i++) {
        for(int j = 0; j < c; j++)
            printf("%d ", C[i][j]);
        printf("\n");
    }

    return 0;
}
